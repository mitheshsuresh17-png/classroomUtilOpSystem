-- Step 1: Start transaction by updating book availability
UPDATE Books
SET available_copies = available_copies - 1 WHERE book_id = 101;

-- Step 2: Set a savepoint after first change
SAVEPOINT after_first_update;

-- Step 3: Another update
UPDATE Books SET available_copies = available_copies - 1 WHERE book_id = 102;

-- Step 4: Oops! Something's wrong with second update, rollback to savepoint
ROLLBACK TO after_first_update;

-- Step 5: Continue with a safe update
UPDATE Books SET available_copies = available_copies + 1 WHERE book_id = 101;

-- Step 6: Commit all changes since savepoint
COMMIT; >
